cmake_minimum_required(VERSION 3.15)
project(projectsockets LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(echo_server src/echo/server.cpp)
add_executable(echo_client src/echo/client.cpp)

# Link Winsock on Windows
if (WIN32)
  target_link_libraries(echo_server PRIVATE Ws2_32)
  target_link_libraries(echo_client PRIVATE Ws2_32)
endif()
